BEGIN
   FOR r IN (SELECT table_name FROM user_tables WHERE table_name IN ('DETALLEBOLETA', 'BOLETA', 'PERSONAL', 'CLIENTE', 'PRODUCTO', 'PROVEEDOR')) LOOP

CREATE TABLE Proveedor (
    idProveedor NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombreP VARCHAR2(100) NOT NULL,
    tipoP VARCHAR2(50) NOT NULL
);

CREATE TABLE Producto (
    idProducto NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombreP VARCHAR2(100) NOT NULL,
    tipoP VARCHAR2(50) NOT NULL,
    precioP NUMBER(10,2) NOT NULL,
    stokP NUMBER NOT NULL,
    idProveedor NUMBER,
    CONSTRAINT fk_prod_prov FOREIGN KEY (idProveedor) REFERENCES Proveedor(idProveedor)
);

CREATE TABLE Cliente (
    idCliente NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombre VARCHAR2(100) NOT NULL,
    telefono VARCHAR2(20)
);

CREATE TABLE Personal (
    idPersonal NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    cargoP VARCHAR2(50) NOT NULL,
    nombreP VARCHAR2(100) NOT NULL,
    hora INTERVAL DAY TO SECOND NOT NULL
);

CREATE TABLE Boleta (
    idB NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    idEmpleadoB NUMBER NOT NULL,
    idClienteB NUMBER NOT NULL,
    FechaB DATE DEFAULT TRUNC(SYSDATE) NOT NULL,
    HoraB INTERVAL DAY TO SECOND NOT NULL,
    CONSTRAINT fk_bol_pers FOREIGN KEY (idEmpleadoB) REFERENCES Personal(idPersonal),
    CONSTRAINT fk_bol_clie FOREIGN KEY (idClienteB) REFERENCES Cliente(idCliente)
);

CREATE TABLE DetalleBoleta (
    idDetalleDB NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    idBoletaDB NUMBER NOT NULL,
    idProductoDB NUMBER NOT NULL,
    cantidadDB NUMBER NOT NULL,
    CONSTRAINT fk_det_bol FOREIGN KEY (idBoletaDB) REFERENCES Boleta(idB),
    CONSTRAINT fk_det_prod FOREIGN KEY (idProductoDB) REFERENCES Producto(idProducto)
);

CREATE OR REPLACE PROCEDURE sp_InsertarCliente(
    p_nombre IN VARCHAR2,

CREATE OR REPLACE PROCEDURE sp_ActualizarCliente(
    p_id IN NUMBER, 
    p_nombre IN VARCHAR2, 
    p_telefono IN VARCHAR2,
    p_resultado OUT VARCHAR2
) AS
BEGIN

    WHEN OTHERS THEN
        p_resultado := '0: Error al actualizar cliente - ' || SQLERRM;
END;

CREATE OR REPLACE PROCEDURE sp_RegistrarVenta(
    p_idEmpleado IN NUMBER,
    p_idCliente IN NUMBER,
    p_detalles IN CLOB,
    p_resultado OUT VARCHAR2,
    p_idBoleta OUT NUMBER
) AS
    v_idBoleta NUMBER;
    v_negativos NUMBER;
BEGIN
    SAVEPOINT inicio_venta;

    INSERT INTO Boleta (idEmpleadoB, idClienteB, FechaB, HoraB)
    VALUES (p_idEmpleado, p_idCliente, TRUNC(SYSDATE), NUMTODSINTERVAL(TO_CHAR(SYSDATE,'SSSSS'),'SECOND'))
    RETURNING idB INTO v_idBoleta;

    INSERT INTO DetalleBoleta (idBoletaDB, idProductoDB, cantidadDB)
    SELECT v_idBoleta, jt.idProducto, jt.cantidad
    FROM JSON_TABLE(p_detalles, '$[*]'
        COLUMNS (
            idProducto NUMBER PATH '$.idProducto',
            cantidad NUMBER PATH '$.cantidad'
        )
    ) jt;

    UPDATE Producto p
    SET p.stokP = p.stokP - (
        SELECT SUM(d.cantidadDB) 
        FROM DetalleBoleta d 
        WHERE d.idBoletaDB = v_idBoleta AND d.idProductoDB = p.idProducto
    )
    WHERE p.idProducto IN (
        SELECT idProductoDB FROM DetalleBoleta WHERE idBoletaDB = v_idBoleta
    );

    SELECT COUNT(*) INTO v_negativos FROM Producto WHERE stokP < 0;
    IF v_negativos > 0 THEN
        ROLLBACK TO inicio_venta;
        p_resultado := '0: Stock insuficiente para realizar la venta';
        p_idBoleta := NULL;
        RETURN;
    END IF;

    COMMIT;
    p_idBoleta := v_idBoleta;
    p_resultado := '1: Venta registrada exitosamente. ID Boleta: ' || v_idBoleta;

EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK TO inicio_venta;
        p_resultado := '0: Error en venta - ' || SQLERRM;
        p_idBoleta := NULL;
END;
/
